🎯 IMPLEMENTAÇÃO RÁPIDA (Para os apressados)
═══════════════════════════════════════════════════════════════════════════════

1. Instale:
   dotnet add src/Fina.Core/Fina.Core.csproj package FluentValidation

2. Crie: src/Fina.Core/Validators/Categorias/CriarCategoriaValidator.cs
   using FluentValidation;
   using Fina.Core.Requests.Categorias;

   namespace Fina.Core.Validators.Categorias;

   public class CriarCategoriaValidator : 
	   AbstractValidator<CriarCategoriaRequest>
   {
	   public CriarCategoriaValidator()
	   {
		   RuleFor(x => x.Titulo)
			   .NotEmpty().WithMessage("Obrigatório")
			   .MinimumLength(3).WithMessage("Min 3")
			   .MaximumLength(80).WithMessage("Max 80");
	   }
   }

3. Registre em src/Fina.Api/Program.cs:
   builder.Services
	   .AddValidatorsFromAssemblyContaining(typeof(Categoria).Assembly)
	   .AddFluentValidationAutoValidation();

4. Pronto! ✅ Seus endpoints funcionam igual.
