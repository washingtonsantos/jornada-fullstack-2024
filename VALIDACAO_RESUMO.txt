PASSO 3: Implementar Validators (20 min)
  // CriarCategoriaValidator.cs
  public class CriarCategoriaValidator : 
	  AbstractValidator<CriarCategoriaRequest>
  {
	  public CriarCategoriaValidator()
	  {
		  RuleFor(x => x.Titulo)
			  .NotEmpty().WithMessage("Obrigatório")
			  .MinimumLength(3).WithMessage("Min 3 caracteres")
			  .MaximumLength(80).WithMessage("Max 80 caracteres");
	  }
  }

PASSO 4: Registrar em Program.cs (2 min)
  builder.Services
	  .AddValidatorsFromAssemblyContaining(typeof(Categoria).Assembly)
	  .AddFluentValidationAutoValidation();

PASSO 5: Limpar Requests (5 min)
  // ANTES
  [Required]
  [StringLength(80, MinimumLength = 3)]
  public string Titulo { get; set; }

  // DEPOIS
  public string Titulo { get; set; }

PASSO 6: Criar Testes (15 min)
  [Fact]
  public void Validate_WithValidData_ReturnsSuccess()
  {
	  var validator = new CriarCategoriaValidator();
	  var result = validator.TestValidate(
		  new() { Titulo = "Alimentação" });
	  result.ShouldNotHaveAnyValidationErrors();
  }

P: Vou quebrar meus endpoints?
R: NÃO! Resposta HTTP será idêntica.

P: Meus Handlers funcionam igual?
R: SIM! ASP.NET Core valida antes de chamar Handler.

P: Preciso instalar em Fina.Api também?
R: Fina.Api precisa de FluentValidation.AspNetCore.

P: Posso validar EF Core async?
R: SIM! Métodos MustAsync() e MustAsyncAsync().

P: Como testo sem Band de datos?
R: TestValidate() não precisa de BD, só logic.

P: FluentValidation é "pesado"?
R: NÃO! É muito mais leve que manter Data Annotations.
